```typescript
function mad() {
  if (Math.random() > 0.5) {
    throw new Error('Random error');
  }
  return 'mad';
}


function foo() {
  let bar: string;
  try {
    bar = mad();
  } catch(err: unknown) {
    // handler error
  }
  console.log(bar)
}
```
